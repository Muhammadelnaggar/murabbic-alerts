    function normSpecies(data){

    data.animalTypeAr || data.animalType || data.species || ''

    function daysBetweenISO(a,b){
    const A = new Date(a), B = new Date(b);
    if (isNaN(A) || isNaN(B)) return NaN;
    A.setHours(0,0,0,0); B.setHours(0,0,0,0);
    return Math.round((B - A)/86400000);
    }

    async function fetchAnimalDocByNumber(number){
    const map = {'٠':'0','١':'1','٢':'2','٣':'3','٤':'4','٥':'5','٦':'6','٧':'7','٨':'8','٩':'9',
    '۰':'0','۱':'1','۲':'2','۳':'3','۴':'4','۵':'5','۶':'6','۷':'7','۸':'8','۹':'9'};
    const raw = String(number || '')
    .trim()
    .replace(/[^\d٠-٩۰-۹]/g,'')
    .replace(/[٠-٩۰-۹]/g, d=>map[d]);
    const asStr = raw;
    const asNum = raw ? Number(raw) : NaN;
    const tries = [
    ['animalNumber','==',asStr],
    ['animalNumber','==',asNum],
    ['number','==',asStr],
    ['number','==',asNum],
    ['animalNo','==',asStr],
    ['animalNo','==',asNum],
    ].filter(t => !(typeof t[2] === 'number' && Number.isNaN(t[2])));
    for (const [field,op,val] of tries){
    const q1 = query(collection(db,'animals'), where(field, op, val), limit(1));
    const snap = await getDocs(q1);
    if (!snap.empty){
    const doc = snap.docs[0];
    return { id: doc.id, data: doc.data() || {} };
    }
    }
    return null;
    }
    async function fetchSignalsFromEvents(number){
    const uid = await getUid();
    const num = String(number||'').trim();
    const qEv = query(
    collection(db,'events'),
    where('userId','==', uid),
    where('animalNumber','==', num),
    orderBy('eventDate','desc'),
    limit(50)
    );
    const snap = await getDocs(qEv);
    let pregnant = null;
    let lastFertileDate = null;
    let lastBoundary = null;
    for (const d of snap.docs){
    const ev = d.data() || {};
    const type = ev.eventType || ev.type || '';
    const res = ev.result || ev.status || '';
    const dt = ev.eventDate;
    if (type === 'ولادة' || type === 'إجهاض'){
    if (!lastBoundary) lastBoundary = dt;
    if (pregnant === null) pregnant = false;
    continue;
    }
    if (type === 'تشخيص حمل'){
    const normRes = String(res).replace(/\s+/g,'').replace(/[ًٌٍَُِّْ]/g,'');
    if (normRes.includes('عشار') && pregnant === null) pregnant = true;
    if (normRes.includes('فارغه') || normRes.includes('فارغة')){
    if (pregnant === null) pregnant = false;
    }
    }
    if (!lastFertileDate && (type === 'تلقيح مخصب' || ev.fertile === true || res === 'مخصب')){
    lastFertileDate = dt;
    }
    if (pregnant !== null && lastFertileDate) break;
    }
    return { pregnant, lastFertileDate, lastBoundary };
    }

    async function guardNotInactive(number){
    const a = await fetchAnimalDocByNumber(number);
    if (!a) return { ok:false, msg:"رقم الحيوان غير موجود في قاعدة البيانات." };
    const st = String(a.data?.status || "").trim().toLowerCase();
    if (st === "inactive") {
    return { ok:false, msg:"❌ لا يمكن تسجيل أي حدث — هذا الحيوان خارج القطيع (بيع/نفوق/استبعاد)." };
    }
    return { ok:true, a };
    }
    async function precheck(number){
    if (SYS_LOCKED) return; // ✅ بعد النجاح لا تعمل أي تحقق جديد تلقائيًا
    disableForm(true);
    showSysbar('جارِ التحقق المبدئي من بيانات الحيوان…');
    const gate = await guardNotInactive(number);
    if (!gate.ok){
    showSysbar(gate.msg, { error:true });
    disableForm(true);
    return;
    }
    const a = gate.a;
    const sp = normSpecies(a.data);
    if (!sp || !MIN_GD[sp]) {
    showSysbar('نوع القطيع غير معروف/غير مدعوم لهذا الحيوان.', { error:true });
    return;
    }
    const rsDoc = a.data.reproductiveStatus || a.data.reproStatus || '';
    const rsNorm = String(rsDoc)
    .replace(/\s+/g,'')
    .replace(/[ًٌٍَُِّْ]/g,'');
    const isPregDoc = /عشار|حامل/i.test(rsNorm);
    const { pregnant: pregEv, lastFertileDate: lfEv, lastBoundary } =
    await fetchSignalsFromEvents(number);
    const isPreg = (pregEv !== null) ? pregEv : isPregDoc;
    const lfDoc = a.data.lastInseminationDate || a.data.lastFertileInseminationDate || '';
    const lf = lfEv || lfDoc;
    if (!isPreg){
    showSysbar('لا يُسمح بتسجيل الولادة: الحالة التناسلية ليست "عشار".', { error:true });
    return;
    }
    if (!lf){
    showSysbar('لا يُسمح بتسجيل الولادة: لا يوجد "آخر تلقيح مُخصِّب".', { error:true });
    return;
    }
    if (lastBoundary && cmpISO(lastBoundary, lf) >= 0){
    showSysbar(`لا يُسمح بتسجيل الولادة: آخر حدث (${lastBoundary}) يلغي أي حمل حالي.`, { error:true });
    return;
    }
    const evISO = (dateEl?.value && dateEl.value.trim()) || todayISO();
    const g = daysBetweenISO(lf, evISO);
    if (!(g >= MIN_GD[sp])){
    offerAbort(number, evISO, g, sp);
    disableForm(false);
    return;
    }
    if (speciesEl) speciesEl.value = sp;
    if (animalIdEl) animalIdEl.value = a.id;
    if (lastFertileEl && !lastFertileEl.value) lastFertileEl.value = lf;
    if (dateEl && !dateEl.value) dateEl.value = evISO;
    showSysbar('جاهز للتسجيل ✅');
    disableForm(false);
    }
